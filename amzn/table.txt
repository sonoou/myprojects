CREATE DATABASE amazon;

CREATE TABLE products_table(
  id VARCHAR(50) PRIMARY KEY,
  image VARCHAR(100),
  priceCents VARCHAR(100)
);

CREATE SEQUENCE cseq INCREMENT BY 1 START WITH 1;
CREATE TABLE customer_table(
  id VARCHAR(50) PRIMARY KEY,
  fname VARCHAR(100),
  lname VARCHAR(100),
  email VARCHAR(100),
  contact varchar(100),
  pass varchar(100)
);

CREATE SEQUENCE cartseq INCREMENT BY 1 START WITH 1;
CREATE TABLE cart_table(
  id VARCHAR(100) PRIMARY KEY,
  customer_id VARCHAR(100),
  FOREIGN KEY (customer_id) REFERENCES customer_table(id) ON DELETE CASCADE ON UPDATE CASCADE,
  product_id VARCHAR(100),
  FOREIGN KEY (product_id) REFERENCES products_table(id) ON DELETE CASCADE ON UPDATE CASCADE,
  quantity VARCHAR(100)
);
